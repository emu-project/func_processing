# Use sphinx to build documentation from docstrings

$ sphinx-quickstart
	use defaults

Edit conf.py (pip install sphinx_rtd_theme if necessary)

	--Path setup
	import os
	import sys
	sys.path.insert(0, os.path.abspath('..'))

	--General configuration
	# extensions = ['sphinx.ext.autodoc']
	extensions = ["sphinx.ext.napoleon"]

	--Options for HTML output
	html_theme = 'sphinx_rtd_theme'

Edit index.rst - add modules

	.. toctree::
	:maxdepth: 2
	:caption: Contents:

	intro
	modules

# $ sphinx-apidoc -o . .. --ext-autodoc
$ sphinx-apidoc -f -o . ..

$ make html
	and "$make html" after any updates to conf.py, index.rst
